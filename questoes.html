async function iniciar() { reportar("Conectando..."); try { const _supabase = supabase.createClient(S_URL, S_KEY); // Buscamos todas as questões para garantir const { data, error } = await _supabase.from('questoes').select('*'); if (error) throw error; const params = new URLSearchParams(window.location.search); const temaAlvo = (params.get('tema') || "").toLowerCase().trim(); document.getElementById('tema-titulo').innerText = params.get('tema') || "Geral"; let filtradas = []; if (temaAlvo) { filtradas = data.filter(q => { // Verifica se o termo alvo (ex: HAS) está contido dentro do tema ou subtema do banco const t = (q.tema || "").toLowerCase(); const s = (q.subtema || "").toLowerCase(); return t.includes(temaAlvo) || s.includes(temaAlvo) || temaAlvo.includes(t); }); } else { filtradas = data; } if (filtradas.length === 0) { reportar("Filtro zerado. Temas encontrados no banco: " + [...new Set(data.map(q => q.tema))].join(", "), true); // BOTÃO DE EMERGÊNCIA: Se não achar HAS, oferece carregar todas as questões document.getElementById('area-questao').innerHTML = `

Não encontramos questões com o nome exato "${temaAlvo}".

`; // Função global para o botão funcionar window.carregarTudo = () => { renderizar(data); document.getElementById('controles').classList.remove('hidden'); reportar("Carregadas todas as questões."); }; } else { reportar("Sucesso: " + filtradas.length + " questões encontradas."); renderizar(filtradas); document.getElementById('controles').classList.remove('hidden'); } } catch (err) { reportar("Erro: " + err.message, true); } }
